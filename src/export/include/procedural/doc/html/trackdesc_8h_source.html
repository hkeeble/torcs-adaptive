148 inline bool TrackDesc::isBetween(int start, int end, int id) {
149  if (start <= end) {
150  if (id >= start && id <= end) { return true; } else { return false; }
151  } else {

157 inline int TrackDesc::diffSegId(int id1, int id2) {
158  int t;
159  if (id1 > id2) { t = id1; id1 = id2; id2 = t; }
160  t = (nTrackSegments - id2 + id1) % nTrackSegments;
161  return MIN(id2 - id1, t);
162 }

165 /* get the segment on which the car is, searching from the position of the last call within range */
166 inline int TrackDesc::getCurrentSegment(tCarElt* car, int lastId, int range)
167 {
168  int start = -(range / 4);
169  int end = range * 3 / 4;
170  double d, min = FLT_MAX;
171  TrackSegment* ts;
172  int minindex = 0;
173 
174  for (int i = start; i < end; i++) {
175  int j = (lastId+i+getnTrackSegments()) % getnTrackSegments();
176  ts = getSegmentPtr(j);
177  d = ts->distToMiddleSqr3D(car->_pos_X, car->_pos_Y, car->_pos_Z);
178  if (d < min) {
179  min = d;
180  minindex = j;
181  }
182  }
183  return minindex;
184 }
