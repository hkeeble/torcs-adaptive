126 /* compute the radius given three points */
127 inline double radius(double x1, double y1, double x2, double y2, double x3, double y3)
128 {
129  double dx1 = x2 - x1;
130  double dy1 = y2 - y1;
131  double dx2 = x3 - x2;
132  double dy2 = y3 - y2;
133 
134  //double z = (dy1*dx2 - dx1*dy2);
135  double z = dx1*dy2 - dy1*dx2;
136 
137  if (z != 0.0) {
138  double k = (dx2*(x3-x1) - dy2*(y1-y3))/z;
139  return sign(z)*sqrt((1.0+k*k)*(dx1*dx1+dy1*dy1))/2.0;
140  } else {
141  return FLT_MAX;
142  }
143 }
